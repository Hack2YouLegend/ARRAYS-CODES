#include <stdio.h>

int main() {
    int n;
    printf("Enter a number: ");
    scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++) {
        printf("Enter element %d: ", i + 1);
        scanf("%d", &arr[i]);
    }
    int sum = arr[0];
    for (int i = 1; i < n; i++) {
        sum -= arr[i];
    }
    printf("MULTIPLE OF NUMBER: %d\n", sum);
    return 0;
}
